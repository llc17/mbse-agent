## SysML v2 语法规范（摘自 OMG SysML v2.0 规范 & Vehicle Example）

标准 part def 写法：
```
part def Resistor {
    doc /* Ideal linear electrical resistor */
    attribute resistance :> ISQ::resistance;
    port p : ElectricalPort;
    port n : ElectricalPort;
}
```

标准 requirement def 写法：
```
requirement def LowPassFilterRequirement {
    doc /* 滤波器应衰减截止频率以上的信号 */
    attribute cutoffFrequency :> ISQ::frequency;
    require constraint { cutoffFrequency == 1.0 / (2.0 * pi * resistance * capacitance) }
}
```

标准 connect 写法：
```
connect r.p to c.p;
connect c.n to ground.g;
```

## 参考示例 1：RC 低通滤波器（电气域）
```
package RCLowPassFilter {{
    import ISQ::*;
    import SI::*;

    port def ElectricalPort;

    part def Resistor {{
        attribute resistance :> ISQ::resistance;
        port p : ElectricalPort;
        port n : ElectricalPort;
    }}

    part def Capacitor {{
        attribute capacitance :> ISQ::capacitance;
        port p : ElectricalPort;
        port n : ElectricalPort;
    }}

    part def Ground {{
        port g : ElectricalPort;
    }}

    part def VoltageSource {{
        attribute voltage :> ISQ::voltage;
        port p : ElectricalPort;
        port n : ElectricalPort;
    }}

    part r : Resistor {{
        attribute redefines resistance = {parameters_R};
    }}

    part c : Capacitor {{
        attribute redefines capacitance = {parameters_C};
    }}

    part gnd : Ground;
    part src : VoltageSource {{
        attribute redefines voltage = 5.0;
    }}

    connect src.p to r.p;
    connect r.n to c.p;
    connect c.n to gnd.g;
    connect src.n to gnd.g;
}}
```

## 参考示例 2：单房间热传导（热域）
```
package SingleRoomThermal {{
    import ISQ::*;
    import SI::*;

    port def ThermalPort;

    part def ThermalCapacitor {{
        attribute heatCapacity :> ISQ::heatCapacity;
        port port_a : ThermalPort;
    }}

    part def ThermalResistor {{
        attribute thermalResistance : Real;
        port port_a : ThermalPort;
        port port_b : ThermalPort;
    }}

    part def FixedTemperature {{
        attribute temperature :> ISQ::temperature;
        port port_a : ThermalPort;
    }}

    part room : ThermalCapacitor {{
        attribute redefines heatCapacity = 500000.0;
    }}

    part wall : ThermalResistor {{
        attribute redefines thermalResistance = 0.02;
    }}

    part outdoor : FixedTemperature {{
        attribute redefines temperature = 308.15;
    }}

    connect outdoor.port_a to wall.port_a;
    connect wall.port_b to room.port_a;
}}
```
